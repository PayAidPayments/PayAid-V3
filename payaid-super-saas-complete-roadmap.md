# PART 5: DATABASE SCHEMA ADDITIONS

## New Models for Website Builder

```prisma
// Website Builder
model Website {
  id String @id
  tenantId String
  tenant Tenant @relation(fields: [tenantId])
  
  name String
  domain String @unique
  sslCertificate String? // For HTTPS
  
  // Design
  template String? // Template ID
  theme Json? // Colors, fonts, etc
  favicon String?
  logo String?
  
  // Analytics
  trackingEnabled Boolean @default(true)
  analyticsId String?
  
  // Status
  isPublished Boolean @default(false)
  publishedAt DateTime?
  
  // Relationships
  pages WebsitePage[]
  visitors WebsiteVisitor[]
  heatmaps Heatmap[]
  sessions SessionRecording[]
  
  createdAt DateTime @default(now())
  updatedAt DateTime @updatedAt
}

model WebsitePage {
  id String @id
  websiteId String
  website Website @relation(fields: [websiteId])
  
  slug String // URL slug
  title String
  description String?
  
  // Content (block-based)
  blocks Json[] // Array of block objects
  
  // Analytics
  pageViews Int @default(0)
  avgTimeOnPage Float @default(0)
  bounceRate Float @default(0)
  
  createdAt DateTime @default(now())
  updatedAt DateTime @updatedAt
  
  @@unique([websiteId, slug])
}

// Website Analytics
model WebsiteVisitor {
  id String @id
  websiteId String
  website Website @relation(fields: [websiteId])
  
  visitorId String // Anonymous ID (tracked via cookie)
  ipAddress String
```
